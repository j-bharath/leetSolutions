class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int front,lagging;
        if(nums.size()<=1)
            return nums.size();
        front=0;lagging=0;
        while(front<nums.size()){
            if(nums[front]!=nums[front+1] || front==nums.size()-1)
                nums[lagging++]=nums[front];
            front++;
        }
        return lagging;
    }
};
